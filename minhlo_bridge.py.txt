import ctypes
import sys
import os

# Tự động nâng cấp đặc quyền Administrator để Windows không chặn tín hiệu bắt phím
def is_admin():
    try: return ctypes.windll.shell32.IsUserAnAdmin()
    except: return False

if not is_admin():
    ctypes.windll.shell32.ShellExecuteW(None, "runas", sys.executable, " ".join(sys.argv), None, 1)
    sys.exit()

import keyboard
import pyperclip
import webbrowser
import time
import re
import pygetwindow as gw
import pyautogui

print("="*60)
print("🚀 ROBOT F9 CLOUD VẬN HÀNH TOÀN VIỆN - BẢN KHÓA CỨNG REFRESH")
print("="*60)

# =====================================================================
# ⚠️ ĐIỀN CHÍNH XÁC ĐƯỜNG LINK TRANG WEB STREAMLIT CLOUD CỦA ANH VÀO ĐÂY:
# =====================================================================
CLOUD_URL = "https://kiem-toan-icd-ksvdphzyqjrms5tm5yhnq6.streamlit.app/" 

last_run_time = 0
cooldown_duration = 2.0

def force_focus_browser():
    try:
        for win in gw.getAllWindows():
            title = win.title.lower()
            if "kiểm toán" in title or "streamlit" in title or "icd-10" in title:
                hwnd = win._hWnd
                if ctypes.windll.user32.IsIconic(hwnd):
                    ctypes.windll.user32.ShowWindow(hwnd, 9) 
                pyautogui.press('alt') 
                ctypes.windll.user32.SetForegroundWindow(hwnd)
                return True
    except Exception:
        pass
    return False

def on_f9_pressed():
    global last_run_time
    current_time = time.time()
    if current_time - last_run_time < cooldown_duration: return
    last_run_time = current_time
    
    pyperclip.copy("") 
    keyboard.send('ctrl+c')
    time.sleep(0.45) # Chờ RAM máy trạm nạp mã bệnh vào bộ nhớ đệm
    
    raw_text = pyperclip.paste().strip().upper()
    copied_text = re.sub(r'[^A-Z0-9.]', '', raw_text)
    
    if not copied_text or len(copied_text) > 10: return
        
    full_url = f"{CLOUD_URL}/?code={copied_text}"

    if force_focus_browser():
        time.sleep(0.25)
        pyautogui.hotkey('ctrl', 'l') 
        time.sleep(0.15)
        pyperclip.copy(full_url)
        time.sleep(0.15)
        pyautogui.hotkey('ctrl', 'v')
        time.sleep(0.15)
        pyautogui.press('enter')
        
        # 💥 ĐOẠN VÁ TỐI THƯỢNG: Ép trình duyệt F5 để nạp mã bệnh mới vào ô nhập liệu đám mây
        time.sleep(0.25)
        pyautogui.press('f5')
        print(f"🔄 Đã ép nạp mã thành công trực tuyến: {copied_text}")
    else:
        webbrowser.open(full_url)
        print("🌐 Khởi tạo điểm liên kết mới thành công!")

keyboard.add_hotkey('f9', on_f9_pressed)
keyboard.wait()

# 1. TỰ ĐỘNG BẬT QUYỀN ADMINISTRATOR ĐỂ TRÁNH BỊ WINDOWS CHẶN BẮT PHÍM
def is_admin():
    try: return ctypes.windll.shell32.IsUserAnAdmin()
    except: return False

if not is_admin():
    ctypes.windll.shell32.ShellExecuteW(None, "runas", sys.executable, " ".join(sys.argv), None, 1)
    sys.exit()

CLOUD_URL = "https://kiem-toan-icd.streamlit.app" 

last_run_time = 0
cooldown_duration = 2.0 # Chốt chặn khóa phím 2 giây triệt tiêu lỗi lặp tab

def force_focus_browser():
    """Thuật toán định vị nâng cao: Quét tìm không gian tab chứa ứng dụng"""
    try:
        for win in gw.getAllWindows():
            title = win.title.lower()
            # Quét tất cả các khả năng hiển thị tiêu đề trên các trình duyệt khác nhau
            if "kiểm toán" in title or "streamlit" in title or "icd-10" in title:
                hwnd = win._hWnd
                if ctypes.windll.user32.IsIconic(hwnd):
                    ctypes.windll.user32.ShowWindow(hwnd, 9) # Khôi phục nếu tab bị thu nhỏ
                pyautogui.press('alt') # Lách luật chặn cửa sổ nổi của Windows
                ctypes.windll.user32.SetForegroundWindow(hwnd)
                return True
    except Exception:
        pass
    return False

def on_f9_pressed():
    global last_run_time
    current_time = time.time()
    if current_time - last_run_time < cooldown_duration: return
    last_run_time = current_time
    
    pyperclip.copy("") 
    keyboard.send('ctrl+c')
    time.sleep(0.45) # Độ trễ đệm RAM tối ưu cho máy trạm bệnh viện nạp mã bệnh
    
    raw_text = pyperclip.paste().strip().upper()
    copied_text = re.sub(r'[^A-Z0-9.]', '', raw_text)
    
    if not copied_text or len(copied_text) > 10:
        print("⚠️ Hủy lệnh: Chuỗi trống hoặc sai cấu trúc mã ICD-10.")
        return
        
    full_url = f"{CLOUD_URL}/?code={copied_text}"

    # Thực thi cơ chế dán đè URL trực tiếp lên thanh địa chỉ (Chống đẻ tab mới)
    if force_focus_browser():
        time.sleep(0.25) # Chờ trình duyệt nổi hẳn lên bề mặt
        pyautogui.hotkey('ctrl', 'l') 
        time.sleep(0.15)
        pyperclip.copy(full_url)
        time.sleep(0.15)
        pyautogui.hotkey('ctrl', 'v')
        time.sleep(0.15)
        pyautogui.press('enter')
        print(f"🚀 Liên thông trực tuyến mã bệnh thành công: {copied_text}")
    else:
        # Nếu chưa bật trang web, khởi tạo tab mới làm mục tiêu gốc
        webbrowser.open(full_url)
        print("🌐 Đã khởi tạo điểm liên kết Đám mây mới!")

keyboard.add_hotkey('f9', on_f9_pressed)
keyboard.wait()

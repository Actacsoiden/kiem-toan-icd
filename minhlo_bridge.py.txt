import keyboard
import pyperclip
import webbrowser
import time
import os
import re
import pygetwindow as gw
import pyautogui

print("="*60)
print("🚀 ROBOT F9 LOCAL - CHẠY NỘI BỘ (KHÔNG CẦN INTERNET)")
print("="*60)

# Cấu hình đường dẫn nội bộ
CURRENT_DIR = os.path.dirname(os.path.abspath(__file__))
SHARED_TXT_PATH = os.path.join(CURRENT_DIR, "shared_code.txt")
LOCAL_URL = "http://localhost:8501"

last_run_time = 0
cooldown_duration = 2.0  # Khóa phím 2 giây chống đẻ tab rác

def on_f9_pressed():
    global last_run_time
    current_time = time.time()
    
    if current_time - last_run_time < cooldown_duration:
        return
    last_run_time = current_time
    
    pyperclip.copy("") 
    keyboard.send('ctrl+c')
    time.sleep(0.4)
    
    raw_text = pyperclip.paste().strip().upper()
    copied_text = re.sub(r'[^A-Z0-9.]', '', raw_text)
    
    if not copied_text or len(copied_text) > 10:
        return
        
    # Ghi mã vào file text để web nội bộ đọc
    try:
        with open(SHARED_TXT_PATH, "w", encoding="utf-8") as f:
            f.write(copied_text)
    except Exception:
        return

    # Tìm trang web đang mở ở localhost và ấn F5
    target_win = None
    for win in gw.getAllWindows():
        if "Kiểm Toán" in win.title or "Streamlit" in win.title or "localhost" in win.title:
            target_win = win
            break
            
    if target_win:
        try:
            if target_win.isMinimized: 
                target_win.restore()
            pyautogui.press('alt') 
            target_win.activate()
            time.sleep(0.3)
            pyautogui.press('f5')
        except Exception:
            webbrowser.open(LOCAL_URL)
    else:
        webbrowser.open(LOCAL_URL)

keyboard.add_hotkey('f9', on_f9_pressed)
keyboard.wait()
